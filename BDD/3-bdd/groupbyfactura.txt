SELECT FACTURA.FAC_COD_VENDEDOR,SUM(FACTURA.FAC_COD_CLIENTE) FROM FACTURA
GROUP BY FACTURA.ID;


SELECT FACTURA.ID AS cliente,
YEAR(FACTURA.FAC_FECHA) AS Año,MONTH(FACTURA.FAC_FECHA) AS Mes,DAY(FACTURA.FAC_FECHA) AS Dia,
SUM(FACTURA.FAC_IMPORTE_NETO)AS Total
FROM FACTURA GROUP BY cliente,Año,Mes,Dia;


SELECT FACTURA.ID AS FACTURA,
YEAR(FACTURA.FAC_FECHA) AS Año,MONTHNAME(FACTURA.FAC_FECHA) AS Mes,DAY(FACTURA.FAC_FECHA) AS Dia,
SUM(FACTURA.FAC_IMPORTE_NETO)AS Total
FROM FACTURA GROUP BY Año,Mes,Dia;



SELECT FACTURA.ID AS FACTURA,
YEAR(FACTURA.FAC_FECHA) AS Año,MONTHNAME(FACTURA.FAC_FECHA) AS Mes,DAY(FACTURA.FAC_FECHA) AS Dia,
MIN(FACTURA.FAC_IMPORTE_NETO)AS Minimo,
MAX(FACTURA.FAC_IMPORTE_NETO)AS Maximo,
AVG(FACTURA.FAC_IMPORTE_NETO)AS Media,
SUM(FACTURA.FAC_IMPORTE_NETO)AS Total,
COUNT(*)AS Operaciones
FROM FACTURA GROUP BY Año,Mes;
